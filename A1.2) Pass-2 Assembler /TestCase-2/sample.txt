START 101
READ N
MOVER BREG ONE
MOVEM BREG TERM
AGAIN MULT BREG TERM
MOVER CREG TERM
ADD CREG ONE
MOVEM CREG TERM
ADD AREG BREG
COMP CREG N
BC LE AGAIN
DIV BREG TWO
MOVEM BREG RESULT
PRINT RESULT
MOVER AREG ='5'
MOVER BREG ='10'
LTORG
STOP
N DS 1
RESULT DS 1
ONE DC 1
TERM DS 1
ORIGIN TERM
TWO DC 2
ONE EQU TWO
END
